  • Random hex color generator 
    
          function randHexColor()
          {
            let color = '#'
            for (i = 0; i<6; i++)
              {
                let values = '0123456789ABCDEF'
                color += values[(Math.floor(Math.random() *16))]
              }
            return color
          }
